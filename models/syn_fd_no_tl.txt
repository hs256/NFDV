program ssd {
  int blist(sip&dip) = 0;
  int threshold = 100;
  int syn_ts(sip&dip) = 0;
  int ftag(sip&dip) = 0;
  float ratio = 0.5;
  rule tcp_syn = flag_syn:1;
  rule tcp_ack = flag_ack:1;

  entry {
    match_flow { f matches tcp_syn; }
    match_state { ftag != 1;}
    action_flow {resubmit;}
    action_state { blist = blist + 1; ftag = 1; } 
  }

  entry {
    match_flow { f matches tcp_syn; }
    match_state { ftag != 1;}
    action_flow { pass; }
    action_state { blist = blist*ratio; } 
  }

  entry {
    match_state { ftag == 1; blist >= threshold }
    action_flow {f[dip] = drop; }
  }

  entry {
    match_state { ftag == 1; blist < threshold }
    action_flow {pass; }
  }
  entry {
    match_flow { f mismatches tcp_syn && f matches tcp_ack; }
    match_state { ftag != 1; blist > 0 }
    action_flow { pass; }
    action_state { blist = blist-1; }
  }

}
